SET function getComputerChoice
    SET choice to random number between 0-2
    IF choice is 0
        RETURN "rock"
    ELSE IF choice is 1
        RETURN "paper"
    ELSE
        RETURN "scissors"
ENDFUNCTION

SET function playRound that takes two string parameters: 
computerSelection and playerSelection
    //tie
    IF computerSelection equals playerSelection
        RETURN "Tie! Both players chose " playerSelection
    //computer wins
    ELSE IF computerSelection is "rock" and playerSelection is "scissors"
    or computerSelection is "paper" and playerSelection is "rock"
    or computerSelection is "scissors" and playerSelection is "paper"
        RETURN "You lose the round! " computerSelection " beats " playerSelection
    //player wins
    ELSE
        RETURN "You win the round! " playerSelection " beats " computerSelection
ENDFUNCTION

SET function getPlayerChoice
    PROMPT user for value of playerChoice (rock, paper, scissors)
    SET playerChoice to given value
    SET playerChoice to all lowercase

    WHILE the input is invalid (i.e. the checkPlayerChoice function returns false)
        PROMPT the user for input again with an additional error message

    RETURN playerChoice
ENDFUNCTION

SET function checkPlayerChoice with an input parameter of the player's choice (playerChoice)
    IF playerChoice is a valid choice (rock, paper, scissors)
        RETURN true
    ELSE
        return false
ENDFUNCTION

SET function playGame
    SET playerScore to 0
    SET computerScore to 0
    INITIALIZE roundResult

    FOR 5 loops
        GET playerChoice
        GET computerChoice
        GET result of the round by calling the playRound function

        OUTPUT result of the round (with outputRoundResults function)

        IF player wins (playRound's 5th letter is "w")
            INCREMENT playerScore
        ELSE IF player loses (playRound's 5th letter is "l")
            INCREMENT computerScore

    COMPARE final scores and outputs game results (with outputGameResults function)
ENDFUNCTION

SET function outputRoundResults with parameters for computerChoice, playerChoice, roundResult, roundCount
    SET output as string to record round number, the computer's choice, the player's choice, and the round result
    RETURN output
ENDFUNCTION

SET function outputGameResults with parameters computerScore and playerScore
    IF playerScore is greater than computerScore
        RETURN "You win the game! You scored " playerScore 
               " and the computer scored " computerScore "."
    ELSE IF playerScore is less than computerScore
        RETURN "You lose the game! You scored " playerScore 
               " and the computer scored " computerScore "."
    ELSE
        RETURN "You tied the game! You both scored " + playerScore + "."
ENDFUNCTION
